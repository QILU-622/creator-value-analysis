核心 SQL

1. 供给漏斗

用于确认问题发生在注册、首发，还是持续经营阶段。

WITH creator_base AS (
    SELECT
        creator_id,
        join_date,
        has_first_publish,
        active_weeks,
        monetization_opened,
        stable_updater,
        CASE WHEN has_first_publish = 1 THEN 1 ELSE 0 END AS published_creator,
        CASE WHEN has_first_publish = 1 AND active_weeks >= 2 THEN 1 ELSE 0 END AS active_2w_creator,
        CASE WHEN has_first_publish = 1 AND active_weeks >= 4 THEN 1 ELSE 0 END AS active_4w_creator
    FROM creator_profile
)
SELECT * FROM creator_base;

2. 关键人群分层

资源占用与长期价值需要分开识别，避免单一总分掩盖错配结构。

WITH creator_scored AS (
    SELECT
        creator_id,
        retention_30d_rate,
        rev_per_1k_exposure,
        total_supported_exposure,
        total_cash_incentive,
        CASE
            WHEN exposure_pct_rank >= 0.60 AND efficiency_pct_rank <= 0.52 THEN '高曝光低变现'
            WHEN retention_pct_rank >= 0.65 AND efficiency_pct_rank >= 0.65 AND resource_pct_rank <= 0.68 THEN '高潜低激励'
            WHEN retention_pct_rank >= 0.72 AND efficiency_pct_rank >= 0.72 THEN '高价值稳定供给'
            WHEN exposure_pct_rank <= 0.25 AND efficiency_pct_rank <= 0.35 THEN '低价值供给'
            ELSE '一般供给'
        END AS creator_segment
    FROM creator_profile
)
SELECT creator_segment, COUNT(*) AS creators
FROM creator_scored
GROUP BY 1;

3. 同规模回测

核心原则是固定 Top N，在同规模下比较新旧规则，避免样本池变化干扰读数。

WITH eligible_creators AS (
    SELECT
        creator_id,
        old_rule_score,
        total_cash_incentive,
        retention_30d_rate,
        unit_incentive_revenue,
        (0.60 * old_rule_pct
         + 0.20 * retention_pct
         + 0.15 * efficiency_pct
         + 0.05 * (1 - resource_pct)) AS new_priority_score
    FROM creator_profile
    WHERE has_first_publish = 1
)
SELECT *
FROM eligible_creators
ORDER BY new_priority_score DESC
LIMIT 4200;

4. 名单迁移结构

用于识别新旧 Top N 之间的重合、补入与移出结构。

WITH old_logic AS (
    SELECT creator_id FROM eligible_creators ORDER BY old_rule_score DESC LIMIT 4200
), new_rule AS (
    SELECT creator_id FROM eligible_creators ORDER BY new_priority_score DESC LIMIT 4200
)
SELECT
    CASE
        WHEN o.creator_id IS NOT NULL AND n.creator_id IS NOT NULL THEN 'retained'
        WHEN o.creator_id IS NULL AND n.creator_id IS NOT NULL THEN 'promoted_into_new_topn'
        WHEN o.creator_id IS NOT NULL AND n.creator_id IS NULL THEN 'dropped_from_old_topn'
        ELSE 'outside_both'
    END AS selection_transition,
    COUNT(*) AS creators
FROM eligible_creators e
LEFT JOIN old_logic o ON e.creator_id = o.creator_id
LEFT JOIN new_rule n ON e.creator_id = n.creator_id
GROUP BY 1;

实验监控

SRM、护栏和扩量 / 回滚判断需要日级可复核的监控查询。

-- 示例：SRM 与护栏日级监控模板
SELECT
    event_date,
    experiment_group,
    COUNT(DISTINCT creator_id) AS creators,
    AVG(retention_30d_rate) AS retention_30d_rate,
    AVG(unit_incentive_revenue) AS unit_incentive_revenue,
    AVG(total_cash_incentive) AS avg_cash_incentive,
    AVG(complaint_rate) AS complaint_rate,
    AVG(cheat_flag) AS cheat_rate
FROM experiment_daily_monitor
GROUP BY 1, 2
ORDER BY 1, 2;
上线前至少要定义：主指标、护栏、SRM 预警阈值、暂停条件、回滚条件，以及对 creator 级分层随机实验的监控口径。
